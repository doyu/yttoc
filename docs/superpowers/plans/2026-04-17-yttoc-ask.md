# yttoc-ask Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `yttoc-ask`, a read-only CLI that answers natural-language questions about a video course using 2 thin tool-use primitives over the yttoc cache.

**Architecture:** Two public functions (`get_summaries`, `get_xscript_range`) wrap existing cache files. A new `ask()` function runs an OpenAI Responses API tool-use loop. The LLM decides what to fetch; Python only formats output. Structured Outputs return `{answer, citations: [{video_id, seconds}]}`.

**Tech Stack:** Python 3.9+, openai SDK (existing dep, Responses API), nbdev 3, fastcore

**Spec:** `docs/superpowers/specs/2026-04-16-yttoc-ask-design.md`

def get_summaries(video_id: str, # Exact video_id
                  root: Path = None # Root cache directory (default: ~/.cache/yttoc)
                 ) -> dict: # summaries.json content verbatim, or {"error": "..."}
    "Return summaries.json for a cached video. No transformation — file content returned as-is."

    sum_path = root / video_id / 'summaries.json'
    if not sum_path.exists():
        return {'error': f'summaries.json not found for {video_id}'}
    return json.loads(sum_path.read_text(encoding='utf-8'))

#| export
def get_xscript_range(video_id: str, # Exact video_id
                      start: int | float, # Start time in seconds
                      end: int | float, # End time in seconds
                      root: str | Path = None # Root cache directory
                     ) -> list[dict] | dict: # [{start, end, text}, ...] or {"error": "..."}
    "Return parsed xscript segments within [start, end). Raw parse_xscript + slice_segments output."
    root = Path(root) if root else _DEFAULT_ROOT
    d = root / video_id
    srt_files = _glob_srt(d)
    if not srt_files:
        return {'error': f'No captions found for {video_id}'}
    segments = parse_xscript(srt_files[0])
    return slice_segments(segments, start, end)
```

#| export
from yttoc.summarize import get_summaries as _read_summaries
from yttoc.xscript import get_xscript_range as _read_xscript_range
from yttoc.core import fmt_duration

def _find_section(sections: list[dict], seconds: int) -> dict | None:
    "Find the section containing the given timestamp. Returns None if no match."
    for s in sections:
        if s['start'] <= seconds < s['end']:
            return s
    return None

def format_citations(citations: list[dict], # [{video_id, seconds}, ...]
                     root: Path = None # Cache root for summaries lookup
                    ) -> list[str]: # Formatted citation lines
    "Resolve [{video_id, seconds}] into display lines with YouTube deep links."
    from yttoc.fetch import _DEFAULT_ROOT
    root = root or _DEFAULT_ROOT
    lines = []
    for i, c in enumerate(citations, 1):
        vid, sec = c['video_id'], c['seconds']
        url = f'https://youtu.be/{vid}?t={sec}'
        ts = fmt_duration(sec)
        sums = _read_summaries(vid, root)
        if 'error' in sums:
            lines.append(f'  [{i}] {vid} @ {ts}\n      {url}')
            continue
        title = sums['video'].get('title', vid)
        section = _find_section(sums['sections'], sec)
        if section:
            lines.append(f'  [{i}] {title} §{section["path"]} "{section["title"]}" @ {ts}\n      {url}')
        else:
            lines.append(f'  [{i}] {title} @ {ts}\n      {url}')
    return lines
```

def _dispatch_tool(name: str, args: dict, root: Path) -> str:
    "Execute a tool call and return the JSON-serialized result."
    if name == 'get_summaries':
        result = _read_summaries(args['video_id'], root)
    elif name == 'get_xscript_range':
        result = _read_xscript_range(args['video_id'], args['start'], args['end'], root)
    else:
        result = {'error': f'Unknown tool: {name}'}
    return json.dumps(result, ensure_ascii=False)

#| export
import openai

def ask(question: str, # Natural-language query
        video_ids: list[str], # Cached video IDs
        model: str = 'gpt-4o', # LLM model
        max_iterations: int = 20, # Safety cap on tool-use loop
        root: Path = None # Cache root directory
       ) -> dict: # {answer: str, citations: [{video_id, seconds}]}
    "Run a tool-use loop to answer a question about a video course."
    from yttoc.fetch import _DEFAULT_ROOT
    root = root or _DEFAULT_ROOT
    client = openai.OpenAI()

    user_input = f"Video IDs: {json.dumps(video_ids)}\n---\nQuestion: {question}"

    response = client.responses.create(
        model=model,
        instructions=SYSTEM_PROMPT,
        input=user_input,
        tools=TOOLS,
        text={"format": ANSWER_SCHEMA},
    )

    for _ in range(max_iterations):
        # Collect function calls from response output
        tool_calls = [item for item in response.output
                      if item.type == 'function_call']
        if not tool_calls:
            break

        # Execute each tool call
        tool_outputs = []
        for tc in tool_calls:
            args = json.loads(tc.arguments)
            print(f'{tc.name}({", ".join(f"{k}={v!r}" for k,v in args.items())})',
                  file=sys.stderr)
            result = _dispatch_tool(tc.name, args, root)
            tool_outputs.append({
                'type': 'function_call_output',
                'call_id': tc.call_id,
                'output': result,
            })

        # Continue conversation with tool results
        response = client.responses.create(
            model=model,
            previous_response_id=response.id,
            input=tool_outputs,
        )

    # Parse the final structured text output
    text_items = [item for item in response.output if item.type == 'text']
    if not text_items:
        return {'answer': 'No answer generated.', 'citations': []}
    return json.loads(text_items[0].text)
```

from fastcore.script import call_parse, Param

@call_parse
def yttoc_ask(question: str, # Natural-language query
              ids: Param('Cached video IDs (1+ required)', str, nargs='+'), # Video IDs
              model: str = 'gpt-4o', # LLM model
              max_iterations: int = 20, # Safety cap on tool-use loop
              root: str = None, # Root cache directory
              verbose: bool = False, # Show tool-use trace on stderr
             ):
    "Answer a question about a video course using LLM tool use."
    from yttoc.fetch import _DEFAULT_ROOT
    root_path = Path(root) if root else _DEFAULT_ROOT

    if not verbose:
        # Suppress stderr tool-call trace by redirecting in ask()
        # For now, verbose is always on (stderr trace is built into ask())
        pass

    result = ask(question, ids, model=model, max_iterations=max_iterations, root=root_path)

    # Output answer
    print(result['answer'])

    # Output resolved citations
    if result['citations']:
        lines = format_citations(result['citations'], root_path)
        print()
        print('Citations:')
        for line in lines:
            print(line)
```
